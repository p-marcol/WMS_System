\chapter{Spis skrótów i symboli}

\begin{itemize}
    \item[HCM] System Zarządzania Kapitałem Ludzkim (ang. \english{Human Capital Management}).
    \item[RFID] Identyfikacja za pomocą fal radiowych (ang. \english{Radio-Frequency Identification}).
    \item[NFC] Komunikacja bliskiego zasięgu (ang. \english{Near Field Communication}).
    \item[LED] Dioda elektroluminescencyjna (ang. \english{Light Emitting Diode}).
    \item[HTTP] Protokół komunikacji w sieci WWW (ang. \english{Hypertext Transfer Protocol}).
    \item[JSON] Notacja obiektów JavaScript (ang. \english{JavaScript Object Notation}).
    \item[MAC] Adres sprzętowy karty sieciowej (ang. \english{Media Access Control address}).
    \item[JPA] Standard mapowania obiektowo-relacyjnego (ang. \english{Java Persistence API}).
    \item[DTO] Obiekt transferu danych (ang. \english{Data Transfer Object}).
    \item[JWT] Token JSON (ang. \english{JSON Web Token}).
\end{itemize}
